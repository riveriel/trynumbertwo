  <script>
    let currentStep = 1;
    let cableOrder = ['red', 'blue', 'green'];
    let connectedCables = [];
    
    function connectCable(color) {
      if (cableOrder[connectedCables.length] === color) {
        connectedCables.push(color);
        document.getElementById(color).classList.add('connected');
        if (connectedCables.length === cableOrder.length) {
          alert("Cables connected successfully!");
          nextStep();
        }
      } else {
        alert("Wrong cable! Try again.");
      }
    }
    
    function replacePiston() {
      alert("Piston replaced!");
      nextStep();
    }
    
    function refillFuel() {
      alert("Fuel tank refilled!");
      nextStep();
    }
    
    function nextStep() {
      document.getElementById(`step${currentStep}`).classList.remove('active');
      currentStep++;
      if (document.getElementById(`step${currentStep}`)) {
        document.getElementById(`step${currentStep}`).classList.add('active');
      } else {
        document.getElementById("success").classList.add('active');
      }
    }
  </script>
